# 清空文件系统缓存, 测试读写命令之前需要执行该命令

# dd 写入速度测试
dd if=/dev/zero of=./test_write count=200 bs=1024k
# dd 读取速度测试
dd if=./test_write of=/dev/null bs=1024k
